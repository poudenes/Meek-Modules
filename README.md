# Meek-Modules

GPIO Schema:

MP1:
  Touch1:       GPIO13
  Dopplet:      GPIO 12
  Output 1:     GPIO 5
  Speaker:      GPIO 15
  RGB Led:      GPIO 2
  Touch Power:  GPIO 16

MT1:
  Touch1:       GPIO 12
  Output 1:     GPIO 5
  Speaker:      GPIO 0
  RGB Led:      GPIO 2
  Touch Power:  GPIO 16

MT2:
  Touch1:       GPIO 12
  Touch2:       GPIO 13     
  Output 1:     GPIO 5
  Output 2:     GPIO 4
  Speaker:      GPIO 0
  RGB Led:      GPIO 2
  Touch Power:  GPIO 16

MT3:
  Touch1:       GPIO 14
  Touch2:       GPIO 12
  Touch3:       GPIO 13
  Output 1:     GPIO 5
  Output 2:     GPIO 4
  Output 3:     GPIO 15
  Speaker:      GPIO 0
  RGB Led:      GPIO 2
  Touch Power:  GPIO 16
